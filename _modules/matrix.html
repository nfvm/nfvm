from nfvm.log import *

from scipy.sparse import coo_matrix, csr_matrix, linalg, triu, tril

class Matrix:
    """稀疏矩阵类型
    
    >>> mat = fvm.Lapacian(T)
    >>> T.value = mat.solve()
    """
    __typeName = "Matrix"
    
    def __init__(self, vf:volScalarField): 
        """稀疏矩阵初始化
        
        Args:
            vf (volScalarField): 体积标量场
        """
        self.vf = vf
        self.N = vf.mesh.cn 
        self.n = len(vf.mesh.neighbour)
        
        self.x = np.zeros(self.N)
        
        self.lowerAddr = vf.mesh.owner[:self.n]
        self.upperAddr = vf.mesh.neighbour # np.zeros(self.n, dtype=int)
        self.diagAddr = np.linspace(0, self.N-1, self.N, dtype=int)
        
        self.b: np.ndarray = np.zeros(self.N)
        self.internalCoeffs: np.ndarray = np.zeros(self.N)
        self.boundaryCoeffs: np.ndarray = np.zeros(self.N)
        
        self.csr = self.__csr()
                
    def __csr(self) -> csr_matrix:
        row = np.concatenate((self.lowerAddr, self.diagAddr, self.upperAddr), 
                             axis=0, dtype=int)
        col = np.concatenate((self.upperAddr, self.diagAddr, self.lowerAddr), 
                             axis=0, dtype=int)
        data = np.zeros(self.n+self.N+self.n)
        return coo_matrix((data, (row, col))).tocsr()
    
    
    def __repr__(self):
        return f"{self.__typeName} with size [{self.N},{self.N}]."

    def upper(self) -> np.ndarray:
        """返回上三角阵数据"""
        return triu(self.csr, k=1, format='csr').data


    
    


[文档]
    def diag(self) -> np.ndarray:
        """返回矩阵对角数据"""
        return self.csr.diagonal(k=0)

    def lower(self) -> np.ndarray:
        """返回下三角阵数据"""
        return tril(self.csr, k=-1, format='csr').data


          
    def __getitem__(self, index):
        return self.csr[index]
    

    def __setitem__(self, index, value):
        self.csr[index] = value

    def __add__(self, other:"Matrix") -> "Matrix":
        """ 重载 + 运算符

        Args:
            other (Matrix): 稀疏方程组类型

        Returns:
            Matrix: return self
        """
        self.internalCoeffs += other.internalCoeffs
        self.boundaryCoeffs += other.boundaryCoeffs
        self.csr += other.csr
        return self


    


[文档]
    def __eq__(self, vf:volScalarField) -> "Matrix":
        """重载 == 运算符

        Args:
            vf (volScalarField): 体积标量场

        Returns:
            Matrix: return self
        """
        self.boundaryCoeffs += vf.value
        return self

    def solve(self, assignment=False) -> np.ndarray:
        """基于scipy.sparse.linalg.solve求解稀疏矩阵

        Returns:
            np.ndarray: 求解结果
        """
        bb = self.b+self.boundaryCoeffs
        mat = self.csr.copy()
        for i in range(self.N):
            mat[i,i] += self.internalCoeffs[i] 
            
        self.x = linalg.spsolve(mat, bb)
        
        if assignment:
            self.vf.value = self.x
            
        return self.x
