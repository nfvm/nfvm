class TriangleMesh:
    """三角网格类，具备以下数据成员：

        1. points (np.array): 点坐标
        2. cells (np.array): 控制体的顶点在 `points` 内的索引，顺时针
        3. faces (np.array): 内部面，面的顶点在 `points` 内的索引
        4. owner (np.array): 内部面所属控制体，持有该面的 `cell` 索引
        5. neighbour (np.array): 内部面相邻控制体，该面相邻的 `cell` 索引
        6. boundary (Boundary): 网格边界面   
        7. pn (int): 点数目
        8. cn (int): 控制体数目
        9. fn (int): 内部面数目
        
    Notes:

        + TriangleMesh 中所有三角形均为顺时针，但不要求输入三角形的顺序，其会自动检测并翻转
        
    """
    __typeName = "TriangleMesh"


    def __init__(self, points: np.ndarray, cells: np.ndarray,
                       faces: np.ndarray, owner: np.ndarray, 
                       neighbour: np.ndarray, boundary: Boundary):
        """初始化 TriangleMesh

        Args:
            points (np.array): 二维点坐标数据
            cells (np.array): 控制体顶点在 `points` 内的索引
        """
        info(f"初始化 {self.__typeName} ...")
        self.points = points 
        self.cells = cells
        self.faces = faces
        self.owner = owner
        self.neighbour = neighbour
        self.boundary: Boundary = boundary

        self.pn: int = self.points.shape[0]      # pointNumber
        self.cn: int = self.cells.shape[0]       # cellNumber
        self.fn: int = self.faces.shape[0]       # internal faceNumber  
        
        self.__cellCenter()
        self.__area()  
        self.__face2cellCenter()
        self.__faceInfo()

        info(f"初始化 {self.__typeName} 成功")
    

    def __repr__(self):
        return f"{self.__typeName} with size {self.cn} cells."
     
     
    def __cellCenter(self):
        self.cell_center = np.zeros((self.cn,2))
        for i in range(self.cn):
            a,b,c = self.cells[i]
            [x1,y1] = self.points[a]
            [x2,y2] = self.points[b]
            [x3,y3] = self.points[c]
            self.cell_center[i,:] = [(x1+x2+x3)/3.0,(y1+y2+y3)/3.0]
            
        self.face_center = np.zeros((self.fn,2))
        for i in range(self.fn):
            a,b = self.faces[i]
            [x1,y1] = self.points[a]
            [x2,y2] = self.points[b]
            self.face_center[i,:] = [(x1+x2)/2.0,(y1+y2)/2.0]
            
        for i in range(self.boundary.size):
            a,b = self.boundary.faces[i]
            [x1,y1] = self.points[a]
            [x2,y2] = self.points[b]
            self.boundary.face_center[i,:] = [(x1+x2)/2.0,(y1+y2)/2.0]

       
    def __area(self):
        """计算网格面积
        """
        self.area: np.ndarray = np.zeros(self.cells.shape[0])
        for i, cell in enumerate(self.cells):
            a,b,c = cell
            [x1, y1] = self.points[a,:]
            [x2, y2] = self.points[b,:]
            [x3, y3] = self.points[c,:]
        
            self.area[i] = -0.5*(x1*y2+x2*y3+x3*y1-x1*y3-x2*y1-x3*y2)


    def __face2cellCenter(self):
        """计算face距离owner cell和neighbour cell的距离
        """
        # 内部面
        self.owner_distance = np.zeros(self.fn)
        self.neighbour_distance = np.zeros(self.fn)
        for i in range(self.fn):
            [pb_index, pc_index] = self.faces[i]
            pb, pc = self.points[pb_index], self.points[pc_index]
            po_index, pn_index = self.owner[i], self.neighbour[i]
            po, pn = self.cell_center[po_index], self.cell_center[pn_index]
            
            self.owner_distance[i] = point2line(po, pb, pc)
            self.neighbour_distance[i] = point2line(pn, pb, pc)
            
        # 边界面
        for i in range(self.boundary.size):
            [pb_index, pc_index] = self.boundary.faces[i]
            pb, pc = self.points[pb_index], self.points[pc_index]
            
            po_index = self.boundary.owner[i]
            po = self.cell_center[po_index]
            self.boundary.owner_distance[i] = point2line(po, pb, pc)
        
        
    def __faceInfo(self):
        """计算内部面和边界面的长度、法向量
        """
        self.face_length = np.zeros(self.fn)
        self.face_normal = np.zeros((self.fn,2))
        for i in range(self.fn):
            [pa_index, pb_index] = self.faces[i]
            pa, pb = self.points[pa_index], self.points[pb_index]
            self.face_length[i] = point2point(pa, pb)
            self.face_normal[i,:] = face_normal(pa, pb)
            
        # 边界面
        for i in range(self.boundary.size):
            [pa_index, pb_index] = self.boundary.faces[i]
            pa, pb = self.points[pa_index], self.points[pb_index]
            self.boundary.face_length[i] = point2point(pa, pb)
            self.boundary.face_normal[i,:] = face_normal(pa, pb)

        
    def __aspect(self) -> np.ndarray:
        """计算网格纵横比

        Returns:
            np.ndarray: 网格纵横比
        """
        aspect = np.zeros(self.cells.shape[0])

        for i,cell in enumerate(self.cells):
            a,b,c = cell
            [x1, y1] = self.points[a,:]
            [x2, y2] = self.points[b,:]
            [x3, y3] = self.points[c,:]

            dx1, dy1 = abs(x2-x1), abs(y2-y1)
            dx2, dy2 = abs(x3-x1), abs(y3-y1)
            dx3, dy3 = abs(x2-x3), abs(y2-y3)
            
            dx = dx1 if (dx1>dx2) else dx2 
            dx = dx  if (dx>dx3)  else dx3 

            dy = dy1 if (dy1>dy2) else dy2 
            dy = dy  if (dy>dy3)  else dy3
            
            aspect[i] = dx/dy if (dx/dy < 1) else dy/dx

        return aspect

    @classmethod
    def points8cells(cls, points: np.ndarray, 
                          cells: np.ndarray) -> TriangleMesh:
        """以点和控制体形式初始化网格

        Args:
            points (np.ndarray): 二维点数组
            cells (np.ndarray): 三角控制体数组

        Returns:
            TriangleMesh: TriangleMesh类型三角网格
        """
        # 检查并调整三角形为顺时针
        for i,cell in enumerate(cells):
            if not clockwiseTriangle(points[cell, :]):
                [a,b,c] = cell
                cells[i,:] = [a,c,b]
        
        info("重构网格拓扑关系 ...")
        face = []
        owner = [] 
        neighbour = []
            
        for i, cell in enumerate(cells):
            a,b,c = cell
            if [b,a] not in face:
                face.append([a,b])
                owner.append(i)
                neighbour.append(-1)
            else:
                n = face.index([b,a])
                neighbour[n] = i
            
            if [c,b] not in face:
                face.append([b,c])
                owner.append(i)
                neighbour.append(-1)
            else:
                n = face.index([c,b])
                neighbour[n] = i
                
            if [a,c] not in face:
                face.append([c,a])
                owner.append(i)
                neighbour.append(-1)
            else:
                n = face.index([a,c])
                neighbour[n] = i

        faces = np.array(face, dtype=np.int32)
        owner = np.array(owner, dtype=np.int32)
        neighbour = np.array(neighbour, dtype=np.int32)

        # 需要对点进行重新排列，并整理相应的owner，使其连续
        bfaces = faces[np.where(neighbour == -1)]
        bowner = owner[np.where(neighbour == -1)]
        boundary = Boundary(bfaces, bowner)

        faces = faces[np.where(neighbour >= 0)]
        owner = owner[np.where(neighbour >= 0)]
        neighbour = neighbour[np.where(neighbour >= 0)]

        info("重构网格拓扑关系 成功")
        
        return cls(points, cells, faces, owner, neighbour, boundary)

    @classmethod
    def fromMeshIO(cls, filename: str, **kwargs) -> TriangleMesh:
        """利用 meshio 从文件加载数据到三角网格

        Args:
            filename (str): 文件地址

        Returns:
            TriangleMesh: 三角网格
        """
        info(f"从 {filename} 加载网格...")
        mesh = meshio.read(filename, **kwargs)
        points = mesh.points[:,:2]
        cells = mesh.cells[-1].data
        return cls.points8cells(points, cells)

    def save(self, filename: str, **kwargs):
        """保存三角网格数据到符合 nfvm 约定的 HDF5 格式文件中

        Args:
            filename (str): 文件地址
        """
        info(f"保存网格信息到文件 {filename} ...")
        with h5py.File(filename,'w') as f:
            f.attrs["nfvm"] = "0.1.0"

            mesh = f.create_group('mesh')
            mesh.attrs['type'] = self.__typeName
            mesh.create_dataset('points', data=self.points)
            mesh.create_dataset('cells', data=self.cells)
            mesh.create_dataset('faces', data=self.faces)
            mesh.create_dataset('owner', data=self.owner)
            mesh.create_dataset('neighbour', data=self.neighbour)
            
            boundary = mesh.create_group('boundary')
            boundary.create_dataset('faces', data=self.boundary.faces)
            boundary.create_dataset('owner', data=self.boundary.owner)
        info("保存网格信息 成功")

    @classmethod
    def load(cls, filename: str, **kwargs) -> TriangleMesh:
        """从符合 nfvm 约定的 HDF5 格式文件加载数据到三角网格

        Args:
            filename (str): 文件地址

        Returns:
            TriangleMesh: 三角网格
        """
        info(f"从文件 {filename} 加载网格信息 ...")
        with h5py.File(filename,'r') as f:
            if 'nfvm' in f.attrs:
                info(f"文件版本：{f.attrs['nfvm']}")
            else:
                warn(f"文件无 nfvm 属性！")

            mesh = f["/mesh"]
            assert cls.__typeName == mesh.attrs['type'], \
                    f"网格类型不匹配，{cls.__typeName} & {mesh.attrs['type']}"
            points = np.array(mesh["points"])
            cells = np.array(mesh["cells"])
            
            faces = np.array(mesh["faces"])
            owner = np.array(mesh["owner"])
            neighbour = np.array(mesh["neighbour"])

            bfaces = np.array(mesh["boundary"]["faces"])
            bowner = np.array(mesh["boundary"]["owner"])
            boundary = Boundary(bfaces, bowner)
            return cls(points, cells, faces=faces, 
                        owner=owner, neighbour=neighbour,
                        boundary=boundary)

    def writeVTK(self, filename: str, aspect = False):
        """通过 meshio 以 VTK 格式写出网格数据

        Args:
            filename (str): 写出文件名
            aspect (bool, optional): 是否包含网格纵横比信息. Defaults to False.
        """
        points = np.zeros((self.points.shape[0],3))
        points[:,:2] = self.points
        
        cell_data = {}
        if aspect:
            cell_data["aspect"] = []
            cell_data["aspect"].append( self.__aspect() )

        info(f"向 {filename} 写入网格数据 ...")

        meshio.write_points_cells(filename, 
                points, 
                cells=[("triangle", self.cells)],
                cell_data=cell_data )
        
        info(f"向 {filename} 写入网格数据 完成")

    def showBoundary(self, pause:bool=False, time:int=5,
                     save:bool=False, 
                     filename:str="boundary.png", 
                     fontsize=9,
                     fontInterval=1,
                     figsize = (5,5),
                     **kwargs) -> None:
        """通过matplotlib绘制并展示边界

        Args:
            pause (bool, optional): 以定时关闭方式展示边界. Defaults to False.
            time (int, optional): 展示时间，秒. Defaults to 5.
            save (bool, optional): 是否保存为文件. Defaults to False.
            filename (str, optional): 保存文件名. Defaults to "boundary.png".
        """
        bpoints = self.points[self.boundary.faces[:,0]]
        x,y = bpoints[:,0], bpoints[:,1]
        index = self.boundary.circleIndex

        _, ax = plt.subplots(figsize=figsize)
        # text
        i = 0
        while(True):
            if(i>=x.shape[0]):
                break
            ax.text(x[i], y[i], i, 
                    fontsize=fontsize,
                    horizontalalignment='center',
                    verticalalignment='center')
            i += fontInterval
        # face
        for c in range(self.boundary.circleNumber):
            s,e = index[2*c], index[2*c+1]
            ax.plot(x[s:e+1], y[s:e+1], alpha=0.75)
    
            ax.text(x[e], y[e], e, 
                    color='r',
                    fontsize=fontsize,
                    horizontalalignment='center',
                    verticalalignment='center')
        
        ax.set_aspect(1)
        
        if save:
            info(f"生成边界图像到 {filename} ...")
            plt.savefig(filename, dpi=300)
            info(f"生成边界图像到 {filename} 完成")
        if pause:
            plt.pause(time)
        else:
            plt.show()
